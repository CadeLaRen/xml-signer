 26 /**
 27  * @name PKCS5PKEY
 28  * @class class for reading passcode protected PKCS#5 private key.
 29  * @description 
 30  * passcode protected PKCS#5 private key reader class. 
 31  * Currently supports only RSA private key and
 32  * following symmetric key algorithms to protect private key.
 33  * <ul>
 34  * <li>DES-EDE3-CBC</li>
 35  * <li>AES-256-CBC</li>
 36  * <li>AES-192-CBC</li>
 37  * <li>AES-128-CBC</li>
 38  * </ul>
 39  * 
 40  */
 41 var PKCS5PKEY = function() {
 42     // *****************************************************************
 43     // *** PRIVATE PROPERTIES AND METHODS *******************************
 44     // *****************************************************************
 45     var decryptAES = function(dataHex, keyHex, ivHex) {
 46 	return decryptGeneral(CryptoJS.AES, dataHex, keyHex, ivHex);
 47     };
 48 
 49     var decrypt3DES = function(dataHex, keyHex, ivHex) {
 50 	return decryptGeneral(CryptoJS.TripleDES, dataHex, keyHex, ivHex);
 51     };
 52 
 53     var decryptGeneral = function(f, dataHex, keyHex, ivHex) {
 54 	var data = CryptoJS.enc.Hex.parse(dataHex);
 55 	var key = CryptoJS.enc.Hex.parse(keyHex);
 56 	var iv = CryptoJS.enc.Hex.parse(ivHex);
 57 	var encrypted = {};
 58 	encrypted.key = key;
 59 	encrypted.iv = iv;
 60 	encrypted.ciphertext = data;
 61 	var decrypted = f.decrypt(encrypted, key, { iv: iv });
 62 	return CryptoJS.enc.Hex.stringify(decrypted);
 63     };
 64 
 65     var getFuncByName = function(algName) {
 66 	return ALGLIST[algName]['proc'];
 67     };
 68 
 69     var ALGLIST = {
 70 	'AES-256-CBC': { 'proc': decryptAES, keylen: 32, ivlen: 16 },
 71 	'AES-192-CBC': { 'proc': decryptAES, keylen: 24, ivlen: 16 },
 72 	'AES-128-CBC': { 'proc': decryptAES, keylen: 16, ivlen: 16 },
 73 	'DES-EDE3-CBC': { 'proc': decrypt3DES, keylen: 24, ivlen: 8 }
 74     };
 75 
 76     var _parsePKCS5PEM = function(sPKCS5PEM) {
 77 	var info = {};
 78 	if (sPKCS5PEM.match(new RegExp("DEK-Info: ([^,]+),([0-9A-Fa-f]+)", "m"))) {
 79 	    info.cipher = RegExp.$1;
 80 	    info.ivsalt = RegExp.$2;
 81 	}
 82 	if (sPKCS5PEM.match(new RegExp("-----BEGIN ([A-Z]+) PRIVATE KEY-----"))) {
 83 	    info.type = RegExp.$1;
 84 	}
 85 	var i1 = -1;
 86 	var lenNEWLINE = 0;
 87 	if (sPKCS5PEM.indexOf("\r\n\r\n") != -1) {
 88 	    i1 = sPKCS5PEM.indexOf("\r\n\r\n");
 89 	    lenNEWLINE = 2;
 90 	}
 91 	if (sPKCS5PEM.indexOf("\n\n") != -1) {
 92 	    i1 = sPKCS5PEM.indexOf("\n\n");
 93 	    lenNEWLINE = 1;
 94 	}
 95 	var i2 = sPKCS5PEM.indexOf("-----END");
 96 	if (i1 != -1 && i2 != -1) {
 97 	    var s = sPKCS5PEM.substring(i1 + lenNEWLINE * 2, i2 - lenNEWLINE);
 98 	    s = s.replace(/\s+/g, '');
 99 	    info.data = s;
100 	}
101 	return info;
102     };
103 
104     var _getKeyAndUnusedIvByPasscodeAndIvsalt = function(algName, passcode, ivsaltHex) {
105 	//alert("ivsaltHex(2) = " + ivsaltHex);
106 	var saltHex = ivsaltHex.substring(0, 16);
107 	//alert("salt = " + saltHex);
108 	    
109 	var salt = CryptoJS.enc.Hex.parse(saltHex);
110 	var data = CryptoJS.enc.Utf8.parse(passcode);
111 	//alert("salt = " + salt);
112 	//alert("data = " + data);
113 
114 	var nRequiredBytes = ALGLIST[algName]['keylen'] + ALGLIST[algName]['ivlen'];
115 	var hHexValueJoined = '';
116 	var hLastValue = null;
117 	//alert("nRequiredBytes = " + nRequiredBytes);
118 	for (;;) {
119 	    var h = CryptoJS.algo.MD5.create();
120 	    if (hLastValue != null) {
121 		h.update(hLastValue);
122 	    }
123 	    h.update(data);
124 	    h.update(salt);
125 	    hLastValue = h.finalize();
126 	    hHexValueJoined = hHexValueJoined + CryptoJS.enc.Hex.stringify(hLastValue);
127 	    //alert("joined = " + hHexValueJoined);
128 	    if (hHexValueJoined.length >= nRequiredBytes * 2) {
129 		break;
130 	    }
131 	}
132 	var result = {};
133 	result.keyhex = hHexValueJoined.substr(0, ALGLIST[algName]['keylen'] * 2);
134 	result.ivhex = hHexValueJoined.substr(ALGLIST[algName]['keylen'] * 2, ALGLIST[algName]['ivlen'] * 2);
135 	return result;
136     };
137 
138     var _decryptKeyB64 = function(privateKeyB64, sharedKeyAlgName, sharedKeyHex, ivsaltHex) {
139 	var privateKeyWA = CryptoJS.enc.Base64.parse(privateKeyB64);
140 	var privateKeyHex = CryptoJS.enc.Hex.stringify(privateKeyWA);
141 	var f = ALGLIST[sharedKeyAlgName]['proc'];
142 	var decryptedKey = f(privateKeyHex, sharedKeyHex, ivsaltHex);
143 	return decryptedKey;
144     };
145     
146     // *****************************************************************
147     // *** PUBLIC PROPERTIES AND METHODS *******************************
148     // *****************************************************************
149     return {
150 	/**
151          * decrypt private key by shared key
152 	 * @name version
153 	 * @memberOf PKCS5PKEY
154 	 * @property {String} version
155 	 * @description version string of PKCS5PKEY class
156 	 */
157 	version: "1.0.0",
158 	/**
159          * decrypt private key by shared key
160 	 * @name getDecryptedKeyHexByKeyIV
161 	 * @memberOf PKCS5PKEY
162 	 * @function
163 	 * @param {String} encryptedKeyHex hexadecimal string of encrypted private key
164 	 * @param {String} algName name of symmetric key algorithm (ex. 'DES-EBE3-CBC')
165 	 * @param {String} sharedKeyHex hexadecimal string of symmetric key
166 	 * @param {String} ivHex hexadecimal string of initial vector(IV).
167 	 * @return {String} hexadecimal string of decrypted privated key
168 	 */
169 	getDecryptedKeyHexByKeyIV: function(encryptedKeyHex, algName, sharedKeyHex, ivHex) {
170 	    var f1 = getFuncByName(algName);
171 	    return f1(encryptedKeyHex, sharedKeyHex, ivHex);
172 	},
173 	/**
174          * parse PEM formatted passcode protected PKCS#5 private key
175 	 * @name parsePKCS5PEM
176 	 * @memberOf PKCS5PKEY
177 	 * @function
178 	 * @param {String} sEncryptedPEM PEM formatted protected passcode protected PKCS#5 private key
179 	 * @return {Hash} hash of key information
180 	 * @description
181          * Resulted hash has following attributes.
182 	 * <ul>
183 	 * <li>cipher - symmetric key algorithm name (ex. 'DES-EBE3-CBC', 'AES-256-CBC')</li>
184 	 * <li>ivsalt - IV used for decrypt. Its heading 8 bytes will be used for passcode salt.</li>
185 	 * <li>type - asymmetric key algorithm name of private key described in PEM header.</li>
186 	 * <li>data - base64 encoded encrypted private key.</li>
187 	 * </ul>
188          *
189 	 */
190         parsePKCS5PEM: function(sPKCS5PEM) {
191 	    return _parsePKCS5PEM(sPKCS5PEM);
192 	},
193 	/**
194          * the same function as OpenSSL EVP_BytsToKey to generate shared key and IV
195 	 * @name getKeyAndUnusedIvByPasscodeAndIvsalt
196 	 * @memberOf PKCS5PKEY
197 	 * @function
198 	 * @param {String} algName name of symmetric key algorithm (ex. 'DES-EBE3-CBC')
199 	 * @param {String} passcode passcode to decrypt private key (ex. 'password')
200 	 * @param {String} hexadecimal string of IV. heading 8 bytes will be used for passcode salt
201 	 * @return {Hash} hash of key and unused IV (ex. {keyhex:2fe3..., ivhex:3fad..})
202 	 */
203 	getKeyAndUnusedIvByPasscodeAndIvsalt: function(algName, passcode, ivsaltHex) {
204 	    return _getKeyAndUnusedIvByPasscodeAndIvsalt(algName, passcode, ivsaltHex);
205 	},
206         decryptKeyB64: function(privateKeyB64, sharedKeyAlgName, sharedKeyHex, ivsaltHex) {
207 	    return _decryptKeyB64(privateKeyB64, sharedKeyAlgName, sharedKeyHex, ivsaltHex);
208         },
209 	/**
210          * decrypt PEM formatted protected PKCS#5 private key with passcode
211 	 * @name getDecryptedKeyHex
212 	 * @memberOf PKCS5PKEY
213 	 * @function
214 	 * @param {String} sEncryptedPEM PEM formatted protected passcode protected PKCS#5 private key
215 	 * @param {String} passcode passcode to decrypt private key (ex. 'password')
216 	 * @return {String} hexadecimal string of decrypted RSA priavte key
217 	 */
218 	getDecryptedKeyHex: function(sEncryptedPEM, passcode) {
219 	    // 1. parse pem
220 	    var info = _parsePKCS5PEM(sEncryptedPEM);
221 	    var publicKeyAlgName = info.type;
222 	    var sharedKeyAlgName = info.cipher;
223 	    var ivsaltHex = info.ivsalt;
224 	    var privateKeyB64 = info.data;
225 	    //alert("ivsaltHex = " + ivsaltHex);
226 
227 	    // 2. generate shared key
228 	    var sharedKeyInfo = _getKeyAndUnusedIvByPasscodeAndIvsalt(sharedKeyAlgName, passcode, ivsaltHex);
229 	    var sharedKeyHex = sharedKeyInfo.keyhex;
230 	    //alert("sharedKeyHex = " + sharedKeyHex);
231 
232 	    // 3. decrypt private key
233             var decryptedKey = _decryptKeyB64(privateKeyB64, sharedKeyAlgName, sharedKeyHex, ivsaltHex);
234 	    return decryptedKey;
235 	},
236 	addAlgorithm: function(functionObject, algName, keyLen, ivLen) {
237 	}
238     };
239 }();
